{\bf 4.30}\\
\begin{enumerate}
	\item 
	$d_{ij}^2=(x_i-x_j)^T(x_i-x_j)=x_i^Tx_i+x_j^Tx_j-2x_i^Tx_j$\\
	$\displaystyle \frac{1}{n}\sum_{i=1}^{n}{d_{ij}^2}=
	\frac{1}{n}\sum_{i=1}^{n}{x_i^Tx_i}+x_j^Tx_j-2\left(\sum_{i=1}^{n}x_i^T\right)x_j$\\
	质心在原点$\Longrightarrow \displaystyle \sum_{i=1}^{n}x_i^T=\mathbf{0} \Longrightarrow
	\frac{1}{n}\sum_{i=1}^{n}{d_{ij}^2}=\frac{1}{n}\sum_{i=1}^{n}{x_i^Tx_i}+x_j^Tx_j$\\
	同理可得$\displaystyle \frac{1}{n}\sum_{j=1}^{n}{d_{ij}^2}=\frac{1}{n}\sum_{j=1}^{n}{x_j^Tx_j}+x_i^Tx_i$\\
	$\displaystyle \frac{1}{n^2}\sum_{i=1}^{n}\sum_{j=1}^{n}{d_{ij}^2}=\frac{1}{n}\sum_{i=1}^{n}\left(\frac{1}{n}\sum_{j=1}^{n}{x_j^Tx_j}+x_i^Tx_i\right)=\frac{2}{n}\sum_{i=1}^{n}{x_i^Tx_i}$\\
	$\displaystyle \Longrightarrow -\frac{1}{2}\left( d_{ij}^2-\frac{1}{n}\sum_{i=1}^nd_{ij}^2-\frac{1}{n}\sum_{j=1}^nd_{ij}^2+\frac{1}{n^2}\sum_{i=1}^{n}\sum_{j=1}^{n}{d_{ij}^2} \right)=x_i^Tx_j$
	
	\item 由第一问知$X^TX$可构造\\
	$X^TX$是实对称矩阵 $\Longrightarrow X^TX$正交相似于实对角阵\\
	令$X^TX=T^TAT$，其中$A$是实对角阵且每个元素是正的，$X$是正交阵\\
	令$A=B^2,X=BT$ 
\end{enumerate}
\qed\\
