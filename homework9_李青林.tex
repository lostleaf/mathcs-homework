{\bf 5.42}\\
	即$p_1=\dfrac{1}{2} \qquad p_2=\dfrac{1}{3} \qquad p_3=\dfrac{1}{6}$\\
	$p_{12}=\dfrac{1}{2}\cdot\dfrac{p_2}{p_1}=\dfrac{1}{3}\qquad 
	p_{13}=\dfrac{1}{2}\cdot\dfrac{p_3}{p_1}=\dfrac{1}{6}\qquad
	p_{11}=1-p_{12}-p_{13}=\dfrac{1}{2}$

	$p_{21}=\dfrac{1}{2}\cdot 1=\dfrac{1}{2}\qquad 
	p_{23}=\dfrac{1}{2}\cdot\dfrac{p_3}{p_2}=\dfrac{1}{4}\qquad
	p_{22}=1-p_{2`}-p_{23}=\dfrac{1}{4}$

	$p_{32}=\dfrac{1}{2}\cdot 1=\dfrac{1}{2}\qquad 
	p_{31}=\dfrac{1}{2}\cdot 1=\dfrac{1}{2}\qquad
	p_{33}=1-p_{32}-p_{31}=0$
\qed\\


{\bf 5.44}\\
	即$p_{00}=\dfrac{1}{2} \qquad p_{01}=0 \qquad p_{10}=0 \qquad p_{11}=\dfrac{1}{2}$
	
	$p_{00,01}=\dfrac{1}{2}\cdot p(x_2=1|~x_1=0)=0 \qquad 
	p_{00,10}=\dfrac{1}{2}\cdot p(x_1=1|~x_2=0)=0$
	
	$p_{01,00}=\dfrac{1}{2}\cdot p(x_2=0|~x_1=0)=\half \qquad  
	p_{01,11}=\dfrac{1}{2}\cdot p(x_1=1|~x_2=1)=\half$
	
	$p_{10,00}=\dfrac{1}{2}\cdot p(x_1=0|~x_2=0)=\half \qquad  
	p_{10,11}=\dfrac{1}{2}\cdot p(x_2=1|~x_1=1)=\half$
	
	$p_{11,01}=\dfrac{1}{2}\cdot p(x_1=0|~x_2=1)=0\qquad 
	p_{11,10}=\dfrac{1}{2}\cdot p(x_2=0|~x_1=1)=0$
\qed\\


{\bf 5.28}\\
	令$r$表示restart value\\
	初始时$\pi_i=(1-r)\pi_jp_{ji}$\\
	设加了$n$个环\\
	$
		\left\{
			\begin{aligned}
				\pi_i&=(1-r)\pi_jp_{ji}+n(1-r)\pi_k\\
				\pi_k&=\dfrac{1-r}{n+1}\pi_i
			\end{aligned}
		\right.
	$
	
	$\Longrightarrow\pi_i=\dfrac{(n+1)}{2nr-nr^2-1}\cdot{}(1-r)\pi_jp_{ji}$\\
	$\displaystyle \lim_{n \to \infty}\pi_i=\dfrac{1}{2r-r^2}\cdot{}(1-r)\pi_jp_{ji}$\\
	即当加点环足够多的时候page rank会变成$\dfrac{1}{2r-r^2}$倍
\qed\\
