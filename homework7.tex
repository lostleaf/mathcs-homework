{\bf 5.1}\\
\begin{enumerate}[a.~]
	\item ~\newline \includegraphics[scale=0.4]{a.jpg}
	\item 
		充分性：\\
		假设二分图存在奇环，设奇环为$v_1\ra{}v_2\ra{}v_3\ra\cdots\ra{}v_m\ra{}v_1$\\
		不妨设$v_1$属于左半集$\Lra v_n$属于左半集$\Lra v_1$属于右半集，矛盾\\
		
		必要性：\\
		假设一个不存在奇环的图不是二分图，则它不可以黑白染色\\
		$\Lra \exists u_s,u_t$\\
		从$u_s$沿$u_s\ra{u_1}\ra{u_2}\ra\cdots\ra{u_n}\ra{u_t}$得$u_t$被染成白色\\
		从$u_s$沿$u_s\ra{v_1}\ra{v_2}\ra\cdots\ra{v_m}\ra{u_t}$得$u_t$被染成黑色\\
		则$u_s\ra{u_1}\ra{u_2}\ra\cdots\ra{u_n}\ra{u_t}\ra{v_m}\ra\cdots\ra{v_1}\ra{u_s}$是奇环，矛盾\\
	\item
		假如最后到每个点的概率收敛，设第$n$步落在$i$点的概率为$P(n,i)$\\
		必有$\forall{n} \sum_{i=1}^{|E|}P(n,i)=1$\\
		$\displaystyle\Lra \exists x, \lim_{n \to \infty}P(n,x)>0$\\
		$\displaystyle\Lra \exists N, \forall n>N P(n,x)>0$\\
		由于该图是二分图，$P(n,x)$要么在$n$为奇数是为零，要么在$n$为偶数时为零，矛盾\\
\end{enumerate}
\qed\\

{\bf 5.9}\\
	不妨将$u$标号为$1$，$v$标号为$n$，令$h(i)$表示$i$到$v$的"hitting time"\\
	$
		\left\{
	   		\begin{aligned}
				h(1)&=1+\dfrac{h(2)}{2}\\
				h(i)&=1+\dfrac{h(i-1)+h(i+1)}{2} \quad (2\leq{i}\leq{n-1}) \\
				h(n)&=0
   			\end{aligned}
   		\right.
	$\\
	解方程得$h(1)=n-1$\\
	即$h_{uv}=n-1$\\
	如果去掉$(u,v)$，变成一条链，$h_{uv}=(n-1)^2$
\qed\\
